import pygame

class projectil(pygame.sprite.Sprite):
    def __init__(self,player):
        super().__init__()
        self.vitess = 10*2
        self.image=pygame.image.load("dossier_du_jeu/image/épée_project.png")
        self. rect=self.image.get_rect()
        self.player=player
        self.rect.x=player.rect.x
        self.rect.y = player.rect.y+140

    def remove(self):
        self.player.all_projectiles.remove(self)
    def left_remove(self):
        self.player.all_projectiles2.remove(self)

    def move_right(self):
        self.rect.x += self.vitess
        for monster in self.player.game.all_monster:
            if  pygame.sprite.spritecollideany(self, self.player.game.all_monster):
                 monster.damage(self.player.attack)
                 self.remove()
            if pygame.sprite.spritecollideany(self, self.player.game.comet_event.all_comet):
                 self.player.pv+=20
                 self.remove()
                 self.left_remove()
        if self.rect.x > 1920:
            self.remove()


    def move_left(self):
        self.image=self.image=pygame.image.load("dossier_du_jeu/image/épée_project_retourné.png")
        self.rect.x -= self.vitess
        for monster in self.player.game.all_monster2:
            if pygame.sprite.spritecollideany(self, self.player.game.all_monster2):
                monster.damage_Left(self.player.attack)
                self.left_remove()
            if pygame.sprite.spritecollideany(self, self.player.game.comet_event.all_comet):
                 self.player.game.all_player.pv+=5
        if self.rect.x <-50:
            self.left_remove()
