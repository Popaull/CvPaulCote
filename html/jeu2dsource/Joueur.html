import pygame

from Projectil1 import projectil



background = pygame.image.load('dossier_du_jeu/image/background.png')
image1_perso = "dossier_du_jeu/image/Chevalier.png"
image2_perso = "dossier_du_jeu/image/Chevalier.png"


class Player(pygame.sprite.Sprite):
    def __init__(self, game):
        super().__init__()
        self.game= game
        self.pv=100
        self.max_pv= 100
        self.attack=34
        self.vitesse = 15
        self.all_projectiles = pygame.sprite.Group()
        self.all_projectiles2 = pygame.sprite.Group()
        self.image1_perso=pygame.image.load("dossier_du_jeu/image/Chevalier.png")
        self.rect = self.image1_perso.get_rect()
        self.valide_image=0
        self.rect.x=800
        self.rect.y=800
        self.vit_jump=1


    def secrenne(self ,screnne, x , y):
        screnne.blit(background,(x,y))
  
    def jump(self):
        if self.rect.y  > 700:
            self.rect.y -=self.vit_jump
        elif self.rect.y <800:
            self.rect.y += self.vit_jump


    def damage(self, montant):
        self.pv -= montant
        if self.pv <= 0:
            self.game.game_over()


    def update_health_bar(self, surface):

        color = (255, 6, 66)
        bar_rect = [self.rect.x + 40, self.rect.y - 20, self.pv, 5]
        back_color = (30, 30, 30)
        back_bar_rect = [self.rect.x + 40, self.rect.y - 20, self.max_pv, 5]
        pygame.draw.rect(surface, back_color, back_bar_rect)
        pygame.draw.rect(surface, color, bar_rect)

    def lunch_project(self):
        self.all_projectiles.add(projectil(self))



    def lunch_project2(self):
        self.all_projectiles2.add(projectil(self))




    def move_up(self):
         if not pygame.sprite.spritecollideany(self, self.game.all_monster):

            self.rect.y -= self.vitesse
    
    def move_down(self):
         if not pygame.sprite.spritecollideany(self, self.game.all_monster):

            self.rect.y += self.vitesse
    
    def move_right(self):
        if not pygame.sprite.spritecollideany(self, self.game.all_monster):

            self.rect.x -= self.vitesse

    def move_left(self):
        if not pygame.sprite.spritecollideany(self, self.game.all_monster2):
            self.rect.x += self.vitesse
