import pygame
from comet import comet

class CometFallEvent :
    def __init__(self,game):
        self.game=game
        self.percent=0
        self.percent_speed=10
        self.all_comet=pygame.sprite.Group()
        self.fall_mode =False


    def add_percent(self):
        self.percent += self.percent_speed/100

    def is_full_loaded(self):
        return  self.percent >= 100

    def meteor_fall(self):
        self.game.score+=30
        for i in range(1,10):
            self.all_comet.add(comet(self))


    def reset_event(self):
        self.percent=0
    def attempt_fall(self):
        if self.is_full_loaded() and len(self.game.all_monster) == 0:

            self.meteor_fall()
            self.fall_mode=True
            print("pluit de kenny")


    def updat_bar(self,surface):
        self.add_percent()


        pygame.draw.rect(surface, (0,0,0),[0,surface.get_height()-1070,surface.get_width(),10] )
        pygame.draw.rect(surface, (255, 10, 10), [0, surface.get_height()-1070, (surface.get_width()/100)*self.percent, 10])
