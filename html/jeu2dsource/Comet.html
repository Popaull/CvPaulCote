import pygame
import random

class comet(pygame.sprite.Sprite):
    def __init__(self,event_comet):
        super().__init__()
        self.event_comet=event_comet
        self.image=pygame.image.load("dossier_du_jeu/image/kenny.png")
        self.image=pygame.transform.scale(self.image,(150,150))
        self.rect=self.image.get_rect()
        self.vitesse=random.randint(5*2,8*2 )
        self.rect.x=random.randint(0,1920)
        self.rect.y=-random.randint(0,800)
        self.damage=20


    def remove(self):
        self.event_comet.all_comet.remove(self)

        if len(self.event_comet.all_comet) == 0:
            self.event_comet.reset_event()
            self.event_comet.game.spawn_monster()
            self.event_comet.game.spawn_left_monster()
    def fall(self):
        self.rect.y += self.vitesse
        if self.rect.y>=900:
            self.remove()
            if len(self.event_comet.all_comet)==0:
                self.event_comet.reset_event()
                self.event_comet.fall_mode=False

        if pygame.sprite.spritecollideany(self,self.event_comet.game.all_player):
            self.event_comet.game.player.damage(self.damage)
            self.remove()
