        #commande de supréssion des message :
        @bot.command(name='del')
        async def delete(ctx, number_of_message:int):
        
            message = await ctx.channel.history(limit=number_of_message + 1).flatten()
            print(str(ctx.author.display_name + " !Del "+str(number_of_message)))
        
            for each_message in message:
                await each_message.delete()
